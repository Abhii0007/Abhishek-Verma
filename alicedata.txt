Question:-   write a program to find factorial of the program

Answers:-


#include <stdio.h>

int main()
{
    int n, i;
    unsigned long long factorial = 1;

    printf("Enter an integer: ");
    scanf("%d",&n);

    // show error if the user enters a negative integer
    if (n < 0)
        printf("Error! Factorial of a negative number doesn't exist.");

    else
    {
        for(i=1; i<=n; ++i)
        {
            factorial *= i;              // factorial = factorial*i;
        }
        printf("Factorial of %d = %llu", n, factorial);
    }

    return 0;
}

Question:-   write a program in cpp to get number and stored in the array and when user type 0 it show the sum of array

Answers:-


#include <iostream>

int main()
{
    int arr[100];
    int n,i,sum=0;
    i = 0;
    
    printf("Enter a number (Type '0' to exit): ");
    scanf("%d",&n);
    
    while(n!=0)
    {
        arr[i] = n;
        i++;
        
        printf("Enter a number (Type '0' to exit): ");
        scanf("%d",&n);
    }
    
    for(int j = 0; j < i; j++)
    {
        sum+=arr[j];
    }
    
    printf("Sum of the array is %d",sum);
    
    return 0;
}

Question:-   make a 2 dimensional array program using cpp

Answers:-



#include <iostream>
using namespace std;

int main() 
{
	//declare 2D array with 3 rows and 4 columns 
	int example[3][4];
	
	//stores numbers in 2D array
	for (int i=0; i<3; i++)
	{
		for (int j=0; j<4; j++)
		{
			example[i][j] = i*j;
		}
	}
	
	//print out numbers in 2D array 
	for (int i=0; i<3; i++)
	{
		for (int j=0; j<4; j++)
		{
			cout << example[i][j] << " ";
		}
		cout << endl;
	}
	return 0;
}
